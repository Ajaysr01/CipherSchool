bool palindrome(string ch,int l, int r){
    
    if(l>=r){
        return true;
    }
    if(ch[l] != ch[r]){
        return false;
    }
    return palindrome(ch,l+1, r-1);
    
}



int main() {
    string ch = "madam";
    int n = ch.length();
    bool res = palindrome(ch, 0, n-1);
    if (res){
        cout<<"true";
    }else {
        cout << "false";
    }
    
}

int power(int x, int n){
    if(n==0){
        return 1;
    }else {
        return x*power(x,n-1);
    }
}

int lastOcc(int i, int arr[], int tar){
    if (i<0){
        return -1;
    }
        
    if(arr[i]==tar){
        return i;
    }else {
        return lastOcc(i-1,arr,tar);
    }
    
}
